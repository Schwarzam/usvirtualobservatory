 24  * This function creates a new votable object, and sets up the state manager for
 25  * that object.
 26  * 
 27  * @param params

 31 function voview(params){
 32 	StateManager = EXANIMO.managers.StateManager;
 33 	
 34 	_rd = new readdata(params);
 35 
 36 	StateManager.onstaterevisit = _rd.restoreState;
 37 	StateManager.initialize();
 38 	originalState = readCookie("voviewReloadState");
 39 	if (originalState) {
 40 		// restore the original state using the cookie value
 41 		eraseCookie("voviewReloadState");
 42 		_rd.restoreState({id: originalState});
 43 	}
 44 
 45 	return _rd;
 46 }

 48 function getTextContent(el) {
 49 	var txt = el.textContent;
 50 	if (txt != undefined) {
 51 		return txt;
 52 	} else {
 53 		return getTCRecurs(el);
 54 	}
 55 }

 57 function getTCRecurs(el) {
 58 	// recursive method to get text content of an element
 59 	// used only if the textContent attribute is not defined (e.g., in Safari)
 60 	var x = el.childNodes;
 61 	var txt = '';
 62 	for (var i=0, node; node=x[i]; i++) {
 63 		if (3 == node.nodeType) {
 64 			txt += node.data;
 65 		} else if (1 == node.nodeType) {
 66 			txt += getTCRecurs(node);
 67 		}
 68 	}
 69 	return txt;
 70 }

 72 function getElementsByClass(searchClass,node,tag) {
 73 	var classElements = new Array();
 74 	if (node == undefined) node = document;
 75 	if (tag == undefined) tag = '*';
 76 	var els = node.getElementsByTagName(tag);
 77 	var elsLen = els.length;
 78 	var pattern = new RegExp("(^|\\s)"+searchClass+"(\\s|$)");
 79 	for (var i = 0, j = 0; i < elsLen; i++) {
 80 		if (pattern.test(els[i].className) ) {
 81 			classElements[j] = els[i];
 82 			j++;
 83 		}
 84 	}
 85 	return classElements;
 86 }

 88 // remove a blank-delimited string sub from string s
 89 // if sub occurs multiple times, all are removed
 90 // also normalizes the string by removing blanks
 91 
 92 function removeSubstring(s,sub) {
 93 	var flist = s.split(' ');
 94 	var glist = [];
 95 	for (var i=0, f; f = flist[i]; i++) {
 96 		if (f && f != sub) {
 97 			glist.push(f);
 98 		}
 99 	}
100 	return glist.join(' ');
101 }

145 // pack form parameters into a GET string
146 
147 function getFormPars(formname) {
148 	if (typeof(formname) == "string") {
149 		var form = document.forms[formname];
150 	} else {
151 		form = formname;
152 	}
153 	if (form == undefined) {
154 		// allow non-existent form
155 		return "";
156 	}
157 	var parlist = [];
158 	for (var i=0; i<form.elements.length; i++) {
159 		var el = form.elements[i];
160 		if (el.tagName == "INPUT") {
161 			var value = encodeURIComponent(el.value);
162 			if (el.type == "text" || el.type == "hidden") {
163 				parlist.push(el.name + "=" + value);
164 			} else if (el.type == "checkbox") {
165 				if (el.checked) {
166 					parlist.push(el.name + "=" + value);
167 				} else {
168 					parlist.push(el.name + "=");
169 				}
170 			} else if (el.type == "radio") {
171 				if (el.checked) {
172 					parlist.push(el.name + "=" + value);
173 				}
174 			}
175 		} else if (el.tagName == "SELECT") {
176 			parlist.push(el.name + "=" + encodeURIComponent(el.options[el.selectedIndex].value));
177 		}
178 	}
179 	return parlist.join("&");
180 }

182 // extract form parameters from a GET string and set form values
183 
184 function setFormPars(formname,getstr) {
185 	if (typeof(formname) == "string") {
186 		var form = document.forms[formname];
187 	} else {
188 		form = formname;
189 	}
190 	var parlist = getstr.split("&");
191 	for (var i=0; i<parlist.length; i++) {
192 		var f = parlist[i].split("=");
193 		if (f.length < 2) {
194 			var name = parlist[i];
195 			var value = "";
196 		} else {
197 			// don't know if embedded '=' can happen, but might as well handle
198 			// it
199 			name = f.shift();
200 			value = decodeURIComponent(f.join("="));
201 		}
202 		var el = form[name];
203 		if (el != null) {
204 			if (el.tagName == "INPUT") {
205 				if (el.type == "checkbox") {
206 					if (value) {
207 						el.checked = true;
208 					} else {
209 						el.checked = false;
210 					}
211 				} else {
212 					// text or hidden element
213 					el.value = value;
214 				}
215 			} else if (el.tagName == "SELECT") {
216 				for (var j=0; j < el.options.length; j++) {
217 					var option = el.options[j];
218 					if (option.value == value) {
219 						option.selected = true;
220 					} else {
221 						option.selected = false;
222 					}
223 				}
224 			} else if (el.length > 0) {
225 				// radio buttons
226 				for (j=0; j < el.length; j++) {
227 					if (el[j].value == value) {
228 						el[j].checked = true;
229 					} else {
230 						el[j].checked = false;
231 					}
232 				}
233 			}
234 		}
235 	}
236 }

238 // pack hash table (dictionary) values into a URI-encoded string
239 
240 function encodeHash(dict) {
241 	var s = [];
242 	for (var p in dict) {
243 		s.push(p + '=' + dict[p]);
244 	}
245 	return encodeURIComponent(s.join("$"));
246 }
247 
248 // unpack hash table from URI-encoded string
249 
250 function decodeHash(value) {
251 	var s = decodeURIComponent(value).split("$");
252 	var dict = {};
253 	for (var i=0; i<s.length; i++) {
254 		var p = s[i];
255 		var f = p.split("=");
256 		if (f.length == 1) {
257 			if (p) dict[p] = undefined;
258 		} else if (f.length == 2) {
259 			dict[f[0]] = f[1];
260 		} else {
261 			var field = f.shift();
262 			dict[field] = f.join("=");
263 		}
264 	}
265 	return dict;
266 }

529 			// set id="selected" for the currently selected view
530 			var el = document.getElementById(newview);
531 			if (el) {
532 				me.setView(el);
533 			} else {
534 				me.setView(newview);
535 			}
536 		}
537 		me.restoringState = false;

776 		if (! selectors) return true;
777 		if (selectors.split) {
778 			// looks like a string

780 		} else if (selectors.length) {
781 			// looks like a list
782 			me.selectedRows = me.selectedRows.concat(selectors);
783 		}
784 		// remove any duplicate selectors from the selection
785 		var uniq = [];
786 		var dict = {};
787 		for (var i=0, selector; i < me.selectedRows.length; i++) {
788 			selector = me.selectedRows[i];

790 				dict[selector] = 1;
791 				uniq.push(selector);
792 			}

795 		uniq.sort();
796 		me.selectedRows = uniq;
797 		return true;

962 	 *            {integer} Page number of the VOTABLE page to be displayed.

965 	this.sort = function(sortColumn, sortOrder, newpage) {
966 		if (me.xml == undefined || me.xslt == undefined) return false;
967 //		if (me.valueXslImported && !me.valueAttributeAdded) me.addValueAttribute();
968 //		if (me.tagRowsImported && !me.tagRowsAttributeAdded) me.tagRowsAttribute();
969 		
970 		me.sortSetup(sortColumn, sortOrder, newpage);
971 		// save state so back button works
972 		me.saveState();
973 		
974 		if (! me.myXslProc) {
975 			// Mozilla/IE XSLT processing using Sarissa
976 			if (!window.XSLTProcessor) return me.noXSLTMessage();
977 
978 			me.myXslProc = new XSLTProcessor();
979 			if ((!me.myXslProc) || (!me.myXslProc.importStylesheet))
980 				return me.noXSLTMessage();

982 			// not a string
983 			me.myXslProc.importStylesheet(me.xslt);
984 		}
985 
986 		// do the transform
987 		me.myXslProc.setParameter(null, "sortOrder", me.sortOrder[me.sortColumn]);
988 		me.myXslProc.setParameter(null, "sortColumn", me.sortColumn);
989 		me.myXslProc.setParameter(null, "page", ""+me.page);
990 		me.myXslProc.setParameter(null, "pageLength", ""+me.pageLength);
991 		me.myXslProc.setParameter(null, "selectAllString", me.selectAllString);

993 		if (me.selectAllLabel) {
994 			me.myXslProc.setParameter(null, "selectAllLabel", me.selectAllLabel);
995 		}
996

998 			me.myXslProc.setParameter(null, "selectedRows", me.selectedRows.join(","));
999 		}
1000 		
1001 		// alert("Render select column is "+me.renderSelectColumn);
1002 		if (me.renderSelectColumn){
1003 			me.myXslProc.setParameter(null, "renderPrefixColumn", 1);
1004 		}else if(me.renderPrefixColumn){
1005 			me.myXslProc.setParameter(null, "renderPrefixColumn", 2);			
1006 		}else{
1007 			me.myXslProc.setParameter(null, "renderPrefixColumn", 0);
1008 		};
1009 		
1010 		// set extra XSLT parameters
1011 		for (var p in me.xslParams) {
1012 			me.myXslProc.setParameter(null, p, me.xslParams[p]);
1013 		}
1014 		
1015 		me.setViewParams();		
1016 
1017 		var xmlDoc = me.getXML();
1018 		var tables = xmlDoc.getElementsByTagName("TABLE");
1019 		
1020 		if (!tables || tables.length <= 0) {
1021 		    me.printErrorDoc(me.output, xmlDoc);
1022 		}else{
1023 			me.displayTable(xmlDoc);
1024 		}
1025 		
1026 		me.applyUserFormats();
1027 
1028 		return false;
1029 	};

1034 		var pchanged = newpage != undefined;
1035 
1036 		if (!sortColumn) {
1037 			sortColumn = me.sortColumn || "";
1038 		}
1039 		if (!sortOrder) {
1040 			if (me.sortToggle && sortColumn == me.sortColumn && (! pchanged)) {
1041 				// toggle sort order
1042 				if (me.sortOrder[sortColumn] == "ascending") {
1043 					sortOrder = "descending";
1044 				} else {
1045 					sortOrder = "ascending";
1046 				}
1047 			} else {
1048 				// restore previous sort order or use default
1049 				sortOrder = me.sortOrder[sortColumn] || "ascending";
1050 			}
1051 		}
1052 		me.sortColumn = sortColumn;
1053 		me.sortOrder[sortColumn] = sortOrder;
1054 		me.sortToggle = true;

1460 function setColumnOrder(table, row) {
1461 	// determine the column order from the table
1462 	var rows = table.tBodies[0].rows;
1463 	var maxcolumns = rows.length-1;
1464 	var order = [];
1465 	for (var i=0; i<rows.length; i++) {
1466 		var classname = rows[i].className || "";
1467 		if (classname.indexOf("separator") >= 0) {
1468 			maxcolumns = i;
1469 		} else {
1470 			// ID for row is 'fieldrow_<number>'
1471 			var f = rows[i].id.split('_');
1472 			order.push(parseInt(f[f.length-1],10));
1473 		}
1474 	}
1475 	_rd.setColumnOrder(maxcolumns, order);
